function showPage(id) {
var pages = document.querySelectorAll(’.page’);
for (var i = 0; i < pages.length; i++) {
pages[i].classList.remove(‘active’);
}
document.getElementById(‘page-’ + id).classList.add(‘active’);
window.scrollTo(0, 0);
}

function stars(n) {
var s = ‘’;
for (var i = 0; i < n; i++) s += ‘★’;
for (var i = n; i < 5; i++) s += ‘☆’;
return s;
}

function renderProvs(data) {
var grid = document.getElementById(‘prov-grid’);
var html = ‘’;
for (var i = 0; i < data.length; i++) {
var p = data[i];
var idx = providers.indexOf(p);
html += ‘<div class="prov-card" onclick="openModal(' + idx + ')">’;
html += ‘<div class="prov-header" style="background:' + p.color + '">’;
html += ‘<div class="prov-avatar">’ + p.init + ‘</div>’;
html += ‘<div class="trial-pill">Trial Available</div></div>’;
html += ‘<div class="prov-body">’;
html += ‘<div class="prov-name">’ + p.name + ‘</div>’;
html += ‘<div class="prov-role">’ + p.role + ‘</div>’;
html += ‘<div class="prov-loc">’ + p.loc + ‘</div>’;
html += ‘<div class="prov-bio">’ + p.bio + ‘</div>’;
html += ‘<div class="prov-stars">’ + stars(p.stars) + ’ (’ + p.reviews + ’ review’ + (p.reviews !== 1 ? ‘s’ : ‘’) + ‘)</div>’;
html += ‘<div class="prov-footer"><div class="prov-price">Trial: <strong>Free</strong></div>’;
html += ‘<button class="btn-book">Book Trial</button></div></div></div>’;
}
grid.innerHTML = html;
}

function setFilter(el, cat) {
var chips = document.querySelectorAll(’.filter-chip’);
for (var i = 0; i < chips.length; i++) chips[i].classList.remove(‘active’);
el.classList.add(‘active’);
if (cat === ‘all’) {
renderProvs(providers);
} else {
var filtered = [];
for (var i = 0; i < providers.length; i++) {
if (providers[i].cat === cat) filtered.push(providers[i]);
}
renderProvs(filtered);
}
}

function doSearch() {
var q = document.getElementById(‘search-q’).value.toLowerCase();
var filtered = [];
for (var i = 0; i < providers.length; i++) {
var p = providers[i];
if (p.name.toLowerCase().indexOf(q) > -1 || p.role.toLowerCase().indexOf(q) > -1 || p.loc.toLowerCase().indexOf(q) > -1) {
filtered.push(p);
}
}
renderProvs(filtered);
}

function openModal(i) {
var p = providers[i];
document.getElementById(‘m-title’).textContent = ’Book Trial with ’ + p.name;
document.getElementById(‘m-sub’).textContent = p.role + ’ · ’ + p.loc;
document.getElementById(‘m-trial’).textContent = p.trial;
document.getElementById(‘modal-form’).style.display = ‘block’;
document.getElementById(‘modal-ok’).style.display = ‘none’;
document.getElementById(‘modal’).classList.add(‘open’);
}

function closeModal() {
document.getElementById(‘modal’).classList.remove(‘open’);
}

function closeModalOutside(e) {
if (e.target === document.getElementById(‘modal’)) closeModal();
}

function submitBooking() {
var name = document.getElementById(‘b-name’).value.trim();
var email = document.getElementById(‘b-email’).value.trim();
if (!name || !email) { alert(‘Please enter your name and email.’); return; }
document.getElementById(‘modal-form’).style.display = ‘none’;
document.getElementById(‘modal-ok’).style.display = ‘block’;
}

var annual = false;
var selPlanName = ‘basic’;

function toggleBilling() {
annual = !annual;
var tog = document.getElementById(‘sp-tog’);
if (annual) tog.classList.add(‘on’); else tog.classList.remove(‘on’);
document.getElementById(‘sp-basic-p’).innerHTML = annual ? ‘$10<span>/mo</span>’ : ‘$15<span>/mo</span>’;
document.getElementById(‘sp-pro-p’).innerHTML = annual ? ‘$25<span>/mo</span>’ : ‘$35<span>/mo</span>’;
updateSpBtn();
}

function selPlan(el, plan) {
var opts = document.querySelectorAll(’.plan-opt’);
for (var i = 0; i < opts.length; i++) opts[i].classList.remove(‘sel’);
el.classList.add(‘sel’);
selPlanName = plan;
updateSpBtn();
}

function updateSpBtn() {
var price = selPlanName === ‘basic’ ? (annual ? ‘$10’ : ‘$15’) : (annual ? ‘$25’ : ‘$35’);
var name = selPlanName === ‘basic’ ? ‘Basic’ : ‘Pro’;
document.getElementById(‘sp-btn’).textContent = ‘Continue with ’ + name + ’ (’ + price + ‘/mo)’;
}

function handleSignup() {
var fname = document.getElementById(‘sp-fname’).value.trim();
var email = document.getElementById(‘sp-email’).value.trim();
var cat = document.getElementById(‘sp-cat’).value;
if (!fname || !email || !cat) { alert(‘Please fill in your name, email, and service category.’); return; }
var ok = document.getElementById(‘sp-ok’);
ok.style.display = ‘block’;
ok.scrollIntoView({behavior:‘smooth’});
}

}
