let currentFilter = ‘all’;

function renderProviders(data) {
const grid = document.getElementById(‘providers-grid’);
grid.innerHTML = data.map((p, i) => `<div class="provider-card" onclick="openBooking(${i})"> <div class="provider-card-header" style="background:${p.color}"> <div class="provider-avatar" style="background:rgba(255,255,255,0.2)">${p.init}</div> <div class="trial-badge-small">Trial Available</div> </div> <div class="provider-card-body"> <div class="provider-name">${p.name}</div> <div class="provider-role">${p.role}</div> <div class="provider-location">${p.location}</div> <div class="provider-bio">${p.bio}</div> <div class="provider-stars">${'★'.repeat(p.stars)}${'☆'.repeat(5-p.stars)} <span style="color:var(--text-light);font-size:0.75rem">(${p.reviews} review${p.reviews!==1?'s':''})</span></div> <div class="provider-footer"> <div class="provider-price">Trial: <strong>Free</strong></div> <button class="btn-book">Book Trial</button> </div> </div> </div>`).join(’’);
}

function setFilter(el, cat) {
document.querySelectorAll(’.filter-chip’).forEach(c => c.classList.remove(‘active’));
el.classList.add(‘active’);
currentFilter = cat;
const filtered = cat === ‘all’ ? providers : providers.filter(p => p.cat === cat);
renderProviders(filtered);
}

function filterProviders() {
const q = document.getElementById(‘search-input’).value.toLowerCase();
const filtered = providers.filter(p =>
p.name.toLowerCase().includes(q) ||
p.role.toLowerCase().includes(q) ||
p.location.toLowerCase().includes(q) ||
p.bio.toLowerCase().includes(q)
);
renderProviders(filtered);
}

// ── BOOKING MODAL ──
function openBooking(index) {
const p = providers[index];
document.getElementById(‘modal-provider-name’).textContent = `Book Trial with ${p.name}`;
document.getElementById(‘modal-provider-role’).textContent = `${p.role} · ${p.location}`;
document.getElementById(‘modal-trial-scope’).textContent = p.trial;
document.getElementById(‘booking-form-state’).style.display = ‘block’;
document.getElementById(‘booking-success’).style.display = ‘none’;
document.getElementById(‘booking-modal’).classList.add(‘open’);
}

function closeModal() {
document.getElementById(‘booking-modal’).classList.remove(‘open’);
}

function submitBooking() {
const name = document.getElementById(‘b-name’).value.trim();
const email = document.getElementById(‘b-email’).value.trim();
if (!name || !email) { alert(‘Please fill in your name and email.’); return; }
document.getElementById(‘booking-form-state’).style.display = ‘none’;
document.getElementById(‘booking-success’).style.display = ‘block’;
}

// ── PRICING ──
let pricingMode = ‘monthly’;
function setPricing(mode) {
pricingMode = mode;
const isAnnual = mode === ‘annual’;
document.getElementById(‘monthly-btn’).className = ‘billing-btn’ + (isAnnual ? ‘’ : ’ active’);
document.getElementById(‘annual-btn’).className = ‘billing-btn’ + (isAnnual ? ’ active’ : ‘’);
document.getElementById(‘p-basic-amount’).innerHTML = isAnnual ? ‘$10<span id="p-basic-period">/mo</span>’ : ‘$15<span id="p-basic-period">/mo</span>’;
document.getElementById(‘p-pro-amount’).innerHTML = isAnnual ? ‘$25<span id="p-pro-period">/mo</span>’ : ‘$35<span id="p-pro-period">/mo</span>’;
}

function handlePricingCTA(plan) {
showPage(‘signup’);
setTimeout(() => {
document.querySelectorAll(’.plan-option’).forEach(o => o.classList.remove(‘selected’));
const options = document.querySelectorAll(’.plan-option’);
if (plan === ‘basic’) options[1].classList.add(‘selected’);
if (plan === ‘pro’) options[2].classList.add(‘selected’);
}, 100);
}

// ── SIGNUP ──
let selectedPlan = ‘basic’;
let billingAnnual = false;

function selectPlan(el, plan) {
document.querySelectorAll(’.plan-option’).forEach(o => o.classList.remove(‘selected’));
el.classList.add(‘selected’);
selectedPlan = plan;
updateSignupBtn();
}

function toggleBilling() {
billingAnnual = !billingAnnual;
const toggle = document.getElementById(‘billing-toggle’);
toggle.classList.toggle(‘on’, billingAnnual);
document.getElementById(‘basic-price’).innerHTML = billingAnnual ? ‘$10<span id="basic-period">/mo</span>’ : ‘$15<span id="basic-period">/mo</span>’;
document.getElementById(‘pro-price’).innerHTML = billingAnnual ? ‘$25<span id="pro-period">/mo</span>’ : ‘$35<span id="pro-period">/mo</span>’;
updateSignupBtn();
}

function updateSignupBtn() {
const prices = { free: ‘$0’, basic: billingAnnual ? ‘$10/mo’ : ‘$15/mo’, pro: billingAnnual ? ‘$25/mo’ : ‘$35/mo’ };
const labels = { basic: `Continue with Basic (${prices.basic}) →`, pro: `Continue with Pro (${prices.pro}) →` };
document.getElementById(‘signup-submit’).textContent = labels[selectedPlan];
}

function handleSignup() {
const fname = document.getElementById(‘sp-fname’).value.trim();
const email = document.getElementById(‘sp-email’).value.trim();
const cat = document.getElementById(‘sp-cat’).value;
if (!fname || !email || !cat) { alert(‘Please fill in your name, email, and service category.’); return; }
if (selectedPlan !== ‘free’) {
// Stripe redirect placeholder
const stripeLinks = {
basic: ‘https://buy.stripe.com/nexfolio-basic’,
pro: ‘https://buy.stripe.com/nexfolio-pro’
};
alert(`Redirecting to secure Stripe payment for the ${selectedPlan.charAt(0).toUpperCase()+selectedPlan.slice(1)} plan.\n\nOnce set up, this will go to: ${stripeLinks[selectedPlan]}\n\nFor now, email us at info@thenexfolio.com to complete your signup!`);
}
document.getElementById(‘sp-success’).style.display = ‘block’;
document.getElementById(‘sp-success’).scrollIntoView({ behavior: ‘smooth’ });
}
